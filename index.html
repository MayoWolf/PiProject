    <script>

        const piFacts = [

            "Pi is an irrational number, meaning it never ends or repeats!",

            "The symbol π was first used in 1706 by William Jones.",

            "NASA uses π to calculate planetary orbits!",

            "The world record for memorizing digits of π is over 70,000 digits!",

            "Pi appears in many formulas in physics, from waves to general relativity!",

            "Ancient Egyptians approximated π as 3.16, very close to its actual value!"

        ];


        function generateFact() {

            const fact = piFacts[Math.floor(Math.random() * piFacts.length)];

            document.getElementById("pi-fact").innerText = fact;

        }


        let correctPiDigits = "3.1415926535";

        function checkDigit() {

            let userInput = document.getElementById("digit-input").value;

            let currentDigits = document.getElementById("pi-sequence").innerText;

            let nextDigit = correctPiDigits[currentDigits.length];

            if (userInput === nextDigit) {

                document.getElementById("pi-sequence").innerText += userInput;

                document.getElementById("digit-result").innerText = "✅ Correct! Keep going!";

            } else {

                document.getElementById("digit-result").innerText = "❌ Oops! Try again.";

            }

            document.getElementById("digit-input").value = "";

        }


        function calculateCircle() {

            let radius = document.getElementById("radius").value;

            if (radius) {

                let circumference = (2 * Math.PI * radius).toFixed(2);

                let area = (Math.PI * radius * radius).toFixed(2);

                document.getElementById("circle-result").innerText = `Circumference: ${circumference}, Area: ${area}`;

            } else {

                document.getElementById("circle-result").innerText = "Please enter a radius!";

            }

        }


        const quizQuestions = [

            { question: "What is the first digit of π after 3.14?", options: ["1", "5", "9"], answer: "1" },

            { question: "Who first used the π symbol?", options: ["Albert Einstein", "William Jones", "Isaac Newton"], answer: "William Jones" },

            { question: "What shape is π related to?", options: ["Square", "Circle", "Triangle"], answer: "Circle" }

        ];


        let score = 0;

        let questionIndex = 0;


        function startQuiz() {

            let quizDiv = document.getElementById("quiz-options");

            let questionObj = quizQuestions[questionIndex % quizQuestions.length];

            document.getElementById("quiz-question").innerText = questionObj.question;

            quizDiv.innerHTML = "";

            questionObj.options.forEach(option => {

                let btn = document.createElement("button");

                btn.innerText = option;

                btn.onclick = () => checkQuizAnswer(option, questionObj.answer);

                quizDiv.appendChild(btn);

            });

            quizDiv.classList.remove("hidden");

            document.getElementById("quiz-button").innerText = "Next Question";

        }


        function checkQuizAnswer(selected, correct) {

            let result = document.getElementById("quiz-result");

            if (selected === correct) {

                result.innerText = "✅ Correct!";

                score++;

            } else {

                result.innerText = "❌ Incorrect!";

            }

            document.getElementById("quiz-score").innerText = score;

            questionIndex++;

        }
